###### Basic Binding
```javascript
var grid = new zsmcJs.grid({
    keyField: 'Id',
    maxWidth: '100%',
    columns: columns //references column settgins
});
grid.GridView.draw($("#grid"));
grid.Provider.bindData(data);
```
###### Paging & Sorting (Local)
```javascript
var grid = new zsmcJs.grid({
    keyField: 'Id',
    maxWidth: '100%',
    columns: columns, 
    viewOptions: {
        actionMode: 'local',
        paging: {
            visible: true,
            rowCount: 5,
            blockSize: 5
        },
        sorting: {
            visible: true
        }
    }
});
grid.GridView.draw($("#grid"));
grid.Provider.bindData(data);
```
###### Paging & Sorting (Remote)
```javascript
var grid2 = new zsmcJs.grid({
    keyField: 'Id',
    width: '100%',
    columns: columns,
    viewOptions: {
        actionMode: 'remote',
        loadUrl: '/api/User/GetUsers/',
        paging: {
            visible: true,
            rowCount: 5,
            blockSize: 5
        },
        sorting: {
            visible: true
        }
    }
});
grid2.GridView.draw($("#grid2"));
```
